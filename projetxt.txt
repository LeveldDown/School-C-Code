void kareIslemleri() {

void cemberIslemleri() {
    double r;
    double PI = 3.141592;
    printf("\n-> Cember Secildi\n");

    do {
        printf("Yaricap (r): ");
        scanf("%lf", &r);
        if (r <= 0) printf("Hata: Yaricap Pozitif Olmalidir.\n");
    } while (r <= 0);
    
    printf("Alan: %.2f\n", PI * r * r);

}
